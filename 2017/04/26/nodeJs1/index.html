nodeJs(一)构建简单的图片上传应用

撰写于 2017-04-26 修改于 2017-04-27 分类 nodeJs学习 标签 node.js

前言：本文没有太多的文字讲解，以代码分析初识nodeJs。

参考资料：
node入门

目标：

  • 浏览器输入http://domain/start时，可以看到一个页面，页面上有一个文件上传的表单。
  • 用户可以选择一个图片并提交表单，随后文件将被上传到http://domain/upload，该页面完成上传后会把图片显示在页面上。

结构：

  1. http服务器
  2. 路由
  3. 请求处理
  4. 返回视图

主文件index.js

var server=require("./server");  //require从外部获取模块的接口
var router=require("./router");
var requestHandlers=require("./requestHandlers");
var handle={};  //对象传值，不同的pathname对应不同的处理请求
handle['/']=requestHandlers.start;
handle['/start']=requestHandlers.start;
handle['/upload']=requestHandlers.upload;
handle['/show']=requestHandlers.show;
server.start(router.route,handle);

http服务器server.js

var http = require("http");  //请求node.js自带的http模块
var url = require("url");
function start(route,handle) {
  function onRequest(request, response) {  //http模块自带的方法
    var pathname=url.parse(request.url).pathname;  //方法如下
    console.log("Request for"+pathname+"received;");
    route(handle,pathname,response,request);
  }
  http.createServer(onRequest).listen(8888);  //http模块函数启动服务器监听8888端口
  console.log("Server has started.");
}
exports.start = start;

url.parse()方法介绍

eg：

这里引入一组概念：exports 和 module.exports，同有导出模块功能，形似但非神似。
每一个node.js执行文件都会自动创建一个module对象，同时，module对象会创建一个exports的属性，初始值是{},
exports=module.exports={}
exports是引用 module.exports的值。exports 被改变的时候，module.exports不会被改变，而模块导出的时候，真正导出的执行是module.exports，而不是exports.

路由router.js

function route(handle,pathname,response,request){
  console.log("about to route a request for"+pathname);
  if(typeof handle[pathname]==='function'){  //检查给定路径对应的请求处理程序是否存在
    handle[pathname](response,request);
  } else {
    console.log("no request handle found for"+pathname);
    response.writeHead(404, {"Content-Type": "text/plain"});  //收到请求发送http状态码和http头的内容类型
    response.write("404 Not found");  //向http相应主体发送文本
    response.end();  //完成响应
  }
}
exports.route=route;

请求处理requestHandlers.js

var querystring=require("querystring"),  //对http请求所带的数据进行解析，一般是问号后面“”=“”这类的数据
    fs=require('fs'),  //file system，文件操作模块
    formidable=require("formidable");  //表单模块
function start(response) {
  console.log("Request handler 'start' was called.");
  var body= '<html>'+
    '<head>'+
    '<meta http-equiv="Content-Type" content="text/html; '+
    'charset=UTF-8" />'+
    '</head>'+
    '<body>'+
    '<form action="/upload" enctype="multipart/form-data" '+
    'method="post">'+
    '<input type="file" name="upload" multiple="multiple">'+
    '<input type="submit" value="Upload file" />'+
    '</form>'+
    '</body>'+
    '</html>';
    response.writeHead(200, {"Content-Type": "text/html"});
    response.write(body);
    response.end();
}
function upload(response,request) {
  console.log("Request handler 'upload' was called.");
  var form=new formidable.IncomingForm();  //创建对象
  form.uploadDir='tmp';  //设置上传文件存放的文件夹，默认为系统的临时文件夹，可以使用fs.rename()改变上传文件的存放路径和文件名 临时文件存放路径，如果没有这段代码可能会报错，因为跨磁盘分区移动的问题
  console.log("about to parse");
  form.parse(request,function(error,fields,files){  //解析request对象
    console.log("parsing done");
    fs.renameSync(files.upload.path,"/tmp/test.png");  //重命名，oldpath-->newpath
    response.writeHead(200, {"Content-Type": "text/html"});
    response.write("received image:<br/>");
    response.write("<img src='/show'>");
    response.end();
  });
}
function show(response, request) {
  console.log("Request handler 'show' was called.");
  fs.readFile("/tmp/test.png", "binary", function(error, file) {  //fs.readfile(文件名，编码，回调函数(err,data))异步读取
    if(error) {
      response.writeHead(500, {"Content-Type": "text/plain"});
      response.write(error + "\n");
      response.end();
    } else {
      response.writeHead(200, {"Content-Type": "image/png"});
      response.write(file, "binary");
      response.end();
    }
  });
}
exports.start = start;
exports.upload = upload;
exports.show=show;
